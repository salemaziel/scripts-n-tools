#!/bin/sh
##run as root

echo "deb https://deb.torproject.org/torproject.org $(lsb_release -c) main
deb-src https://deb.torproject.org/torproject.org $(lsb_release -c) main" | sudo tee /etc/apt/sources.list.d/onions.list

curl https://deb.torproject.org/torproject.org/A3C4F0F979CAA22CDBA8F512EE8CBC9E886DDD89.asc | gpg --import

gpg --export A3C4F0F979CAA22CDBA8F512EE8CBC9E886DDD89 | apt-key add -

apt update
apt install tor deb.torproject.org-keyring

sudo add-apt-repository ppa:micahflee/ppa
sudo apt update
sudo apt install -y onionshare torbrowser-launcher
